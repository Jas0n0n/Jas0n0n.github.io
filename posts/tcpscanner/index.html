TcpScanner Using Golang

1.0 基础版的非并发版扫描

package main

import(
	"fmt"
  "net"
)

func main(){
  for i:=1;i<=1024;i++{
    address :=fmt.Sprintf("127.0.0.1:%d",i)
    conn,err:=net.Dial("tcp",address)
    if err!=nil{
      //端口已关闭或过滤
      continue
    }
    conn.Close()
    fmt.Printf("%d is open\n",i)
  }
}

2.0 WaitGrop并发控制版扫描

package main

import (
  "fmt"
  "net"
  "sync"
)

func main(){
  var wg sync.WaitGroup
  for i:=0;i<1024;i++{
    wg.Add(1)
    go func(j int){
      defer wg.Done
      address :=fmt.Sprintf("127.0.0.1:%d",j)
      conn,err :=net.Dial("tcp",address)
      if err !=nil{
        return 
      }
    }(i)
  }
  wg.Wait()
}
